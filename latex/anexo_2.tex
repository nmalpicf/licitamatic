\newcommand{\monthOfYear}[1]
{
  \ifthenelse{\equal{#1}{01}}{Enero}{}
  \ifthenelse{\equal{#1}{02}}{Febrero}{}
  \ifthenelse{\equal{#1}{03}}{Marzo}{}
  \ifthenelse{\equal{#1}{04}}{Abril}{}
  \ifthenelse{\equal{#1}{05}}{Mayo}{}
  \ifthenelse{\equal{#1}{06}}{Junio}{}
  \ifthenelse{\equal{#1}{07}}{Julio}{}
  \ifthenelse{\equal{#1}{08}}{Agosto}{}
  \ifthenelse{\equal{#1}{09}}{Septiembre}{}
  \ifthenelse{\equal{#1}{10}}{Octubre}{}
  \ifthenelse{\equal{#1}{11}}{Noviembre}{}
  \ifthenelse{\equal{#1}{12}}{Diciembre}{}
}
